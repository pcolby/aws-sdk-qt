31 /**
32  * @class AwsAbstractRequest
33  *
34  * @brief Interface class for providing AWS requests
35  *
36  * Typically, a user would pass a derived class (such as SqsCreateQueueRequest)
37  * to an AwsAbstractClient-derived class (such as SqsClient) to send a request
38  * to an AWS service.
39  *
40  * Note, by design, requests based on this class are endpoint-agnostic. That is,
41  * you may instantiate a request (such as an SqsCreateQueueRequest), and pass it
42  * to multiple clients (such as a US instance of SqsClient as well as an AU
43  * instance of SqsClient). Thus it is the caller (typically an
44  * AwsAbstractClient-derived class) that is responsible for for setting the
45  * region / endpoint information, allowing requests to be reused unmodified
46  * multiple times, and/or in multiple regions.
47  */

133 /**
134  * @brief Construct a QNetworkRequest for this AWS request.
135  *
136  * This base implementation used the protected unsignedRequest function to fetch
137  * the basic request to be sent, then uses \a signature and \a credentials to
138  * sign the request before returning to the caller.
139  *
140  * Derived classes may override this if their specific AWS service required more
141  * advanced logic (or disallow signing, for example).
142  *
143  * @param endpoint AWS endpoint the request is to be for.
144  * @param signature AWS signature class for signing the request.
145  * @param credentials AWS credentials to use for signing the request.
146  *
147  * @return A valid, signed (where appropriate) request that can be sent to via
148  * a network access manager to perform the configured request.
149  *
150  * @see unsignedRequest
151  */
152 QNetworkRequest AwsAbstractRequest::networkRequest(
153  const QUrl &endpoint,
154  const AwsAbstractSignature &signature,
155  const AwsAbstractCredentials &credentials) const
156 {
157  QNetworkRequest request(unsignedRequest(endpoint));
158  signature.sign(credentials, operation(), request, data());
159  return request;
160 }

178 /**
179  * @brief Send this request to a given Amazon endpoint, via the specified
180  * network access manager.
181  *
182  * @param manager Network access manager to send the request via.
183  * @param endpoint AWS service endpoint to send the request to.
184  * @param signature Signature implementation to use to sign the request.
185  * @param credentials AWS credentials to use to sign the request.
186  *
187  * @return AwsAbstractResponse-derived object representing the AWS service's response.
188  */
189 AwsAbstractResponse * AwsAbstractRequest::send(QNetworkAccessManager &manager,
190  const QUrl &endpoint,
191  const AwsAbstractSignature &signature,
192  const AwsAbstractCredentials &credentials) const
193 {
194  Q_ASSERT(isValid());
195  Q_D(const AwsAbstractRequest);
196  const QNetworkRequest request(networkRequest(endpoint, signature, credentials));
197  qDebug() << "Request URL:" << request.url().toString(QUrl::FullyEncoded);
198  if (request.hasRawHeader("Authorization")) {
199  qDebug() << "Authorization:" << request.rawHeader("Authorization");
200  }
201  switch (operation()) {
202  case QNetworkAccessManager::DeleteOperation:
203  return response(manager.deleteResource(request));
204  case QNetworkAccessManager::HeadOperation:
205  return response(manager.head(request));
206  case QNetworkAccessManager::GetOperation:
207  return response(manager.get(request));
208  case QNetworkAccessManager::PostOperation:
209  return response(d->post(manager, request));
210  case QNetworkAccessManager::PutOperation:
211  return response(d->put(manager, request));
212  case QNetworkAccessManager::CustomOperation: // Fall through.
213  default:
214  // Catch this in debug mode for easier development / debugging.
215  Q_ASSERT_X(false, Q_FUNC_INFO, "invalid operation");
216  }
217  return NULL;
218 }
