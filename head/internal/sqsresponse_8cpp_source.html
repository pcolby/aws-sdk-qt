64 /**
65  * @brief Get this response's error string.
66  *
67  * This override checks for SQS service errors, and if found, returns the first
68  * one. Otherwise, this override delegates to the base implementation (which
69  * checks for network and XML parse errors, for example).
70  *
71  * Also consider using the serviceErrors function to retrieve all of the servive
72  * errors (if any), not just the first one (though often SQS will only return
73  * one error per response).
74  *
75  * @return An error string, or a null QString if this response has no errors.
76  *
77  * @see AwsAbstractResponse::errorString
78  * @see hasError
79  * @see serviceErrors
80  */
81 QString SqsResponse::errorString() const
82 {
83  Q_D(const SqsResponse);
84  return (d->errors.empty()) ? AwsAbstractResponse::errorString() : d->errors.first().message();
85 }

121 /**
122  * @brief Parse a failure response.
123  *
124  * This implementation parses SQS ErrorResponse elements, as defined by
125  * http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
126  *
127  * For example:
128  * @code{xml}
129  * <ErrorResponse>
130  * <Error>
131  * <Type>Sender</Type>
132  * <Code>AccessDenied</Code>
133  * <Message>Access to the resource http://sqs.us-east-1.amazonaws.com/ is denied.</Message>
134  * <Detail/>
135  * </Error>
136  * <Error>
137  * ...
138  * </Error>
139  * <RequestId>9a285199-c8d6-47c2-bdb2-314cb47d599d</RequestId>
140  * </ErrorResponse>
141  * @endcode
142  *
143  * @param response Response to parse.
144  *
145  * @note Unlike most other SQS responses, the SQS ErrorResponse does not wrap
146  * the RequestId element in a ResponseMetadata element.
147  *
148  * @see http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
149  */
150 void SqsResponse::parseFailure(QIODevice &response)
151 {
152  Q_D(SqsResponse);
153  QXmlStreamReader xml(&response);
154  if (xml.readNextStartElement()) {
155  if (xml.name() == QLatin1String("ErrorResponse")) {
156  d->parseErrorResponse(xml);
157  } else {
158  qWarning() << "ignoring" << xml.name();
159  xml.skipCurrentElement();
160  }
161  }
162  setXmlError(xml);
163 }

186 /**
187  * @brief Parse an SQS ErrorResponse element.
188  *
189  * This implementation parses SQS ErrorResponse elements, as defined by
190  * http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
191  *
192  * For example:
193  * @code{xml}
194  * <ErrorResponse>
195  * <Error>
196  * <Type>Sender</Type>
197  * <Code>AccessDenied</Code>
198  * <Message>Access to the resource http://sqs.us-east-1.amazonaws.com/ is denied.</Message>
199  * <Detail/>
200  * </Error>
201  * <Error>
202  * ...
203  * </Error>
204  * <RequestId>9a285199-c8d6-47c2-bdb2-314cb47d599d</RequestId>
205  * </ErrorResponse>
206  * @endcode
207  *
208  * @param xml XML stream to parse.
209  *
210  * @note Unlike most other SQS responses, the SQS ErrorResponse does not wrap
211  * the RequestId element in a ResponseMetadata element.
212  *
213  * @see http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
214  */
215 void SqsResponsePrivate::parseErrorResponse(QXmlStreamReader &xml)
216 {
217  Q_ASSERT(xml.name() == QLatin1String("ErrorResponse"));
218  while (xml.readNextStartElement()) {
219  if (xml.name() == QLatin1String("Error")) {
220  errors.append(SqsError(xml));
221  } else if (xml.name() == QLatin1String("RequestId")) {
222  requestId = xml.readElementText();
223  } else {
224  qWarning() << "ignoring" << xml.name();
225  xml.skipCurrentElement();
226  }
227  }
228 }

244  *

248  * ResponseMetadata element is common to most (if not all) successful
249  * SQS responses (its simply not used in the SQS ErrorResponse element).
250  *

252  */
