84 /**
85  * @brief Get this response's error string.
86  *
87  * This base implementation returns either the internal network reply object's
88  * error string (if the reply object has an error), or the internal XML parse
89  * error (if one was set, eg via set setXmlError), otherwise a null QString.
90  *
91  * Derived classes may override this function to support additional error types.
92  * Typically such derived implementations would check this base implementation's
93  * result first, or fallback to this base implementation. For example:
94  *
95  * @code
96  * if (weHaveOurOwnCustomError()) {
97  * return customErrorString();
98  * } else {
99  * return AwsAbstractResponse::errorString();
100  * }
101  * @endcode
102  *
103  * @return An error string, or a null QString if this response has no errors.
104  *
105  * @see hasError
106  * @see networkError
107  * @see xmlParseError
108  */
109 QString AwsAbstractResponse::errorString() const
110 {
111  Q_D(const AwsAbstractResponse);
112  if (networkError() != QNetworkReply::NoError) {
113  return d->reply->errorString();
114  } else if (xmlParseError() != QXmlStreamReader::NoError) {
115  return xmlParseErrorString();
116  } else {
117  return QString();
118  }
119 }

121 /**
122  * @brief Did any error occurr processing this reponse?
123  *
124  * Errors could be anything from network requests errors, to errors validating
125  * AWS response content.
126  *
127  * This base implementation supports network, and XML parsing errors. Derived
128  * classes may override this function to support additional error types.
129  *
130  * @return \c true if this response contains one or more errors, \c false
131  * otherwise.
132  *
133  * @note This function may be equivalent to !isValid, however, it is not safe
134  * for this function's implementation to be just that, since derived
135  * classes may assume the inverse, resulting in an infinite cycle between
136  * the two functions.
137  *
138  * @see errorString
139  * @see isValid
140  */
141 bool AwsAbstractResponse::hasError() const
142 {
143  Q_D(const AwsAbstractResponse);
144  return (((d->reply) && (d->reply->error() != QNetworkReply::NoError)) ||
145  (d->xmlError != QXmlStreamReader::NoError));
146 }

149  * @brief Is this AWS response valid?
150  *
151  * This base implementation simply checks for the absense of network and XML
152  * parsing errors. Derived classes may override this function to check other
153  * error types, and/or apply additional validation rules.
154  *

224 /**
225  * @brief Convert an XML stream to a hierarchical QVariantMap.
226  *
227  * This function is used internally to embed opaque XML structures, such as the
228  * SQS service's ErrorResponse::Error::Detail, which the SQS schema defines as
229  * an arbitrary complex type.
230  *
231  * @note This static function exists within the AwsAbstractResponse for
232  * historic reasons. It should probably be moved our of this class, and
233  * into a more generic utility space at some point.
234  *
235  * @param xml XML stream to convert.
236  * @param prefix Prefix to apply to special (non-element) child entry names.
237  * @param maxDepth Maximum depth to traverse before truncating the tree.
238  *
239  * @return A QVariantMap representing the \a xml XML fragment.
240  *
241  * @todo Move this toVariant function to somewhere more generic.
242  */
243 QVariantMap AwsAbstractResponse::toVariant(
244  QXmlStreamReader &xml, const QString &prefix, const int maxDepth)
245 {
246  if (maxDepth < 0) {
247  qWarning() << QObject::tr("max depth exceeded");
248  return QVariantMap();
249  }
250 
251  if (xml.hasError()) {
252  qWarning() << xml.errorString();
253  return QVariantMap();
254  }
255 
256  if (xml.tokenType() == QXmlStreamReader::NoToken)
257  xml.readNext();
258 
259  if ((xml.tokenType() != QXmlStreamReader::StartDocument) &&
260  (xml.tokenType() != QXmlStreamReader::StartElement)) {
261  qWarning() << QObject::tr("unexpected XML tokenType %1 (%2)")
262  .arg(xml.tokenString()).arg(xml.tokenType());
263  return QVariantMap();
264  }
265 
266  QVariantMap map;
267  if (xml.tokenType() == QXmlStreamReader::StartDocument) {
268  map.insert(prefix + QLatin1String("DocumentEncoding"), xml.documentEncoding().toString());
269  map.insert(prefix + QLatin1String("DocumentVersion"), xml.documentVersion().toString());
270  map.insert(prefix + QLatin1String("StandaloneDocument"), xml.isStandaloneDocument());
271  } else {
272  if (!xml.namespaceUri().isEmpty())
273  map.insert(prefix + QLatin1String("NamespaceUri"), xml.namespaceUri().toString());
274  foreach (const QXmlStreamAttribute &attribute, xml.attributes()) {
275  QVariantMap attributeMap;
276  attributeMap.insert(QLatin1String("Value"), attribute.value().toString());
277  if (!attribute.namespaceUri().isEmpty())
278  attributeMap.insert(QLatin1String("NamespaceUri"), attribute.namespaceUri().toString());
279  if (!attribute.prefix().isEmpty())
280  attributeMap.insert(QLatin1String("Prefix"), attribute.prefix().toString());
281  attributeMap.insert(QLatin1String("QualifiedName"), attribute.qualifiedName().toString());
282  map.insertMulti(prefix + attribute.name().toString(), attributeMap);
283  }
284  }
285 
286  for (xml.readNext(); (!xml.atEnd()) && (xml.tokenType() != QXmlStreamReader::EndElement)
287  && (xml.tokenType() != QXmlStreamReader::EndDocument); xml.readNext()) {
288  switch (xml.tokenType()) {
289  case QXmlStreamReader::Characters:
290  case QXmlStreamReader::Comment:
291  case QXmlStreamReader::DTD:
292  case QXmlStreamReader::EntityReference:
293  map.insertMulti(prefix + xml.tokenString(), xml.text().toString());
294  break;
295  case QXmlStreamReader::ProcessingInstruction:
296  map.insertMulti(prefix + xml.processingInstructionTarget().toString(),
297  xml.processingInstructionData().toString());
298  break;
299  case QXmlStreamReader::StartElement: {
300  const QString elementName = xml.name().toString();
301  map.insertMulti(elementName, toVariant(xml, prefix, maxDepth-1));
302  break;
303  }
304  default:
305  qWarning() << QObject::tr("unexpected XML tokenType %1 (%2)")
306  .arg(xml.tokenString()).arg(xml.tokenType());
307  }
308  }
309  return map;
310 }

312 /**
313  * @brief Does a network reply indicate a successful response?
314  *
315  * This base implementation simply checks that \a reply has no errors, and that
316  * it represents an HTTP response with a 2xx HTTP status code.
317  *
318  * Derived classes may override this function to perform other checks, and/or to
319  * support non-HTTP responses.
320  *
321  * This function is used by the base implementation of the parse function, to
322  * delegate responsibility to either the parseSuccess or parseFailure functions.
323  *
324  * @param reply Network reply to check for success.
325  *
326  * @return \c true if \a reply indicates a successful response, \c false otherwise.
327  *
328  * @see parse
329  */
330 bool AwsAbstractResponse::isSuccess(QNetworkReply * const reply) const
331 {
332  return ((reply->error() == QNetworkReply::NoError) &&
333  ((reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt() / 100) == 2));
334 }

388 /**
389  * @brief Parse an AWS response from a network reply.
390  *
391  * This base implementation reports networks, if any occurred, otherwise uses
392  * the result of the virtual isSuccess function to delegate parsing to either
393  * the pure virtual parseSuccess or pure virtual parseFailure methods.
394  *
395  * Typically derived classes will only implement parseSuccess and parseFailure,
396  * without needing to override this function.
397  *
398  * @param reply Network reply from which to parse the AWS response.
399  *
400  * @see isSuccess
401  * @see parseSuccess
402  * @see parseFailure
403  */
404 void AwsAbstractResponse::parse(QNetworkReply * const reply)
405 {
406  if (reply->error() != QNetworkReply::NoError) {
407  qWarning() << reply->errorString();
408  } else if (isSuccess(reply)) {
409  parseSuccess(*reply);
410  } else {
411  parseFailure(*reply);
412  }
413  emit finished();
414 }
