127 /**
128  * @brief Send an AWS request.
129  *
130  * This base implementation simply uses the \a request's \c send method,
131  * supplying the necessary resources (ie the credentials, network access
132  * manager, and signature object) from this object as required.
133  *
134  * Derived classes would not normally need to override this method.
135  *
136  * @param request The AWS request to send.
137  *
138  * @return An object representing the AWS service's response.
139  */
140 AwsAbstractResponse * AwsAbstractClient::send(const AwsAbstractRequest &request)
141 {
142  Q_D(AwsAbstractClient);
143  Q_ASSERT(d->credentials);
144  Q_ASSERT(d->networkAccessManager);
145  Q_ASSERT(d->signature);
146  Q_ASSERT(request.isValid());
147 
148  if (!d->credentials) {
149  qWarning() << "credentials not set";
150  return NULL;
151  } else if (!d->networkAccessManager) {
152  qWarning() << "network access manager not set";
153  return NULL;
154  } else if (!d->signature) {
155  qWarning() << "signature not set";
156  return NULL;
157  } else if (!request.isValid()) {
158  qWarning() << serviceName() << "request not valid";
159  return NULL;
160  }
161 
162  return request.send(*d->networkAccessManager, endpoint(),
163  *d->signature, *d->credentials);
164 }

220 /**
221  * @brief Set the network access manager for this AWS service object.
222  *
223  * This object will not take ownership of the supplied \a manager pointer.
224  * It is the caller's responsibility to ensure that the pointer remains valid
225  * for the life of this object, or until setNetworkAccessManager is used to set
226  * a new pointer.
227  *
228  * @param manager Network access manager for this AWS service object to use.
229  */
230 void AwsAbstractClient::setNetworkAccessManager(QNetworkAccessManager * const manager)
231 {
232  Q_D(AwsAbstractClient);
233 
234  // If \a manager is already our network access manager, drop out early.
235  if (d->networkAccessManager == manager) {
236  return;
237  }
238 
239  // If the old network access manager was our parent, disown our parent.
240  if ((d->networkAccessManager != NULL) && (parent() == d->networkAccessManager)) {
241  setParent(NULL);
242  }
243 
244  // Set the network access manager.
245  d->networkAccessManager = manager;
246 }
