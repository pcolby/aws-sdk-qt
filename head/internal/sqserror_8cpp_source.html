292 /**
293  * @brief Parse an SQS Error element.
294  *
295  * This function parses Error XML elements (ie from SQS ErrorResponse elements)
296  * as specified in http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
297  *
298  * For example:
299  *
300  * @code{xml}
301  * <Error>
302  * <Type>Sender</Type>
303  * <Code>AccessDenied</Code>
304  * <Message>Access to the resource http://sqs.us-east-1.amazonaws.com/ is denied.</Message>
305  * <Detail/>
306  * </Error>
307  * @endcode
308  *
309  * @param xml XML stream to parse.
310  *
311  * @see http://queue.amazonaws.com/doc/2012-11-05/QueueService.wsdl
312  */
313 void SqsErrorPrivate::parse(QXmlStreamReader &xml)
314 {
315  if ((xml.tokenType() == QXmlStreamReader::NoToken) && (!xml.readNextStartElement())) {
316  return;
317  }
318  Q_ASSERT(xml.name() == QLatin1String("Error"));
319  while (xml.readNextStartElement()) {
320  if (xml.name() == QLatin1String("Type")) {
321  type = typeFromString(rawType = xml.readElementText());
322  } else if (xml.name() == QLatin1String("Code")) {
323  code = codeFromString(rawCode = xml.readElementText());
324  } else if (xml.name() == QLatin1String("Message")) {
325  message = xml.readElementText();
326  } else if (xml.name() == QLatin1String("Detail")) {
327  detail = AwsAbstractResponse::toVariant(xml);
328  } else {
329  qWarning() << "ignoring" << xml.name();
330  xml.skipCurrentElement();
331  }
332  }
333 }

342 SqsError::ErrorCode SqsErrorPrivate::codeFromString(const QString &code)
343 {
344  #define IfStringReturnErrorResponse(str) \
345  if (code == QLatin1String(#str)) return SqsError::str
346  IfStringReturnErrorResponse(AccessDenied);
347  IfStringReturnErrorResponse(IncompleteSignature);
348  IfStringReturnErrorResponse(InternalFailure);
349  IfStringReturnErrorResponse(InvalidAction);
350  IfStringReturnErrorResponse(InvalidClientTokenId);
351  IfStringReturnErrorResponse(InvalidParameterCombination);
352  IfStringReturnErrorResponse(InvalidParameterValue);
353  IfStringReturnErrorResponse(InvalidQueryParameter);
354  IfStringReturnErrorResponse(MalformedQueryString);
355  IfStringReturnErrorResponse(MissingAction);
356  IfStringReturnErrorResponse(MissingAuthenticationToken);
357  IfStringReturnErrorResponse(MissingParameter);
358  IfStringReturnErrorResponse(OptInRequired);
359  IfStringReturnErrorResponse(RequestExpired);

361  IfStringReturnErrorResponse(Throttling);
