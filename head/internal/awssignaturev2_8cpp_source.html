57 void AwsSignatureV2::sign(const AwsAbstractCredentials &credentials, const QNetworkAccessManager::Operation operation,
58  QNetworkRequest &request, const QByteArray &data) const
59 {
60  Q_UNUSED(data) // Not included in V2 signatures.
61  Q_D(const AwsSignatureV2);
62 
63  // Set the AccessKeyId, SignatureMethod, SignatureVersion and Timestamp query items, if not already.
64  d->adornRequest(request, credentials);
65 
66  // Calculate the signature.
67  const QByteArray stringToSign = d->canonicalRequest(operation, request.url());
68  const QString signature = QString::fromUtf8(QUrl::toPercentEncoding(QString::fromUtf8(
69  QMessageAuthenticationCode::hash(stringToSign, credentials.secretKey().toUtf8(),
70  d->hashAlgorithm).toBase64())));
71 
72  // Append the signature to the request.
73  QUrl url = request.url();
74  url.setQuery(url.query() + QLatin1String("&Signature=") + signature);
75  request.setUrl(url);
76 }

111 /**
112  * @internal
113  *
114  * @brief Add AWS Signature Version 2 adornments to an AWS request.
115  *
116  * In addition to service-specific request parameters, Amazon requires that version
117  * 2 signatures contain a number of common query parameters. This functions adds
118  * those query parameters to \a request if they're not already present.
119  *
120  * The query parameters added by this function, as required by Amazon, are:
121  * * `AWSAccessKeyId` - set to \a credentials.accessKeyId().
122  * * `SignatureMethod` - set to `HMAC-SHA1` or `HMAC-SHA256`.
123  * * `SignatureVersion` - set to `2`.
124  * * `Timestamp` - set to a current UTC timestamp in an ISO 8601 format, like
125  * `2013-10-30T12:34:56Z`.
126  *
127  * @param request Request to adorn.
128  * @param credentials Credentials to use when adorning \a request.
129  *
130  * @see signatureMethod
131  * @see http://docs.aws.amazon.com/general/latest/gr/signature-version-2.html
132  */
133 void AwsSignatureV2Private::adornRequest(QNetworkRequest &request,
134  const AwsAbstractCredentials &credentials) const
135 {
136  // Set / add the necessary query items.
137  QUrl url = request.url();
138  QUrlQuery query(url);
139  setQueryItem(query, QLatin1String("AWSAccessKeyId"), credentials.accessKeyId());
140  setQueryItem(query, QLatin1String("SignatureVersion"), QLatin1String("2"));
141  setQueryItem(query, QLatin1String("SignatureMethod"), QString::fromUtf8(signatureMethod(hashAlgorithm)));
142  setQueryItem(query, QLatin1String("Timestamp"),
143  QString::fromUtf8(QUrl::toPercentEncoding(
144  QDateTime::currentDateTimeUtc().toString(QLatin1String("yyyy-MM-ddThh:mm:ssZ"))
145  )),
146  false); // Don't warn if its already set to something else.
147 
148  // If we've touched the query items (likely), then update the request.
149  if (query != QUrlQuery(url)) {
150  qDebug() << url;
151  url.setQuery(query);
152  qDebug() << url;
153  request.setUrl(url);
154  }
155 }

157 /**
158  * @internal
159  *
160  * @brief Create an AWS V2 Signature canonical request.
161  *
162  * This function creates a canonical representation of an AWS request as defined by
163  * Amazon's V2 signature specification.
164  *
165  * For example, for the following HTTP `GET` request:
166  *
167  * https://elasticmapreduce.amazonaws.com?Action=DescribeJobFlows&Version=2009-03-31&AWSAccessKeyId=AKIAIOSFODNN7EXAMPLE&SignatureVersion=2SignatureMethod=HmacSHA256Timestamp=2011-10-03T15%3A19%3A30
168  *
169  * this function will return the following canonical form:
170  *
171  * GET
172  * elasticmapreduce.amazonaws.com
173  * /
174  * AWSAccessKeyId=AKIAIOSFODNN7EXAMPLE&Action=DescribeJobFlows&SignatureMethod=HmacSHA256&SignatureVersion=2&Timestamp=2011-10-03T15%3A19%3A30&Version=2009-03-31
175  *
176  * @note All URL components are encoded to UTF-8, as required by Amazon.
177  *
178  * @param operation The HTTP method being requested.
179  * @param url The URL being request.
180  *
181  * @return An AWS V2 Signature canonical request.
182  *
183  * @see http://docs.aws.amazon.com/general/latest/gr/signature-version-2.html
184  */
185 QByteArray AwsSignatureV2Private::canonicalRequest(const QNetworkAccessManager::Operation operation,
186  const QUrl &url) const
187 {
188  return httpMethod(operation).toUtf8() + '\n' +
189  url.host().toUtf8() + '\n' +
190  canonicalPath(url).toUtf8() + '\n' +
191  canonicalQuery(QUrlQuery(url));
192 }

194 /**
195  * @brief Create an AWS V2 Signature method designation.
196  *
197  * This function returns a signature method designation, as defined by Amazon, for
198  * use with V2 signatures.
199  *
200  * For example, if the algorith is `QCryptographicHash::Sha256`, this function will
201  * return `HmacSHA256`.
202  *
203  * @note Amazon only supports two algorithms for V2 signatures - SHA1 and SHA256.
204  *
205  * @param algorithm The hash algorithm to get the canonical designation for.
206  *
207  * @return An AWS V2 Signature method designation.
208  *
209  * @see http://docs.aws.amazon.com/general/latest/gr/signature-version-2.html
210  */
211 QByteArray AwsSignatureV2Private::signatureMethod(const QCryptographicHash::Algorithm algorithm) const
212 {
213  switch (algorithm) {
214  case QCryptographicHash::Sha1: return "HmacSHA1";
215  case QCryptographicHash::Sha256: return "HmacSHA256";
216  default:
217  Q_ASSERT_X(false, Q_FUNC_INFO, "invalid algorithm");
218  return "invalid-algorithm";
219  }
220 }
